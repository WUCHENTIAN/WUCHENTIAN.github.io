##### 1.2. ```vector```（方便）

```c++
#include <cmath>
vector<int> addVec(vector<int> v1, vector<int> v2){
  /**
  * 注意这里的vector将原有的大数倒序存储
  */
  vector<int> ans(max(v1.size(), v2.size()), 0);
  for(int i = 0; i < max(v1.size(), v2.size()); i++){
    if(i < v1.size()){
      ans[i] += v1[i];
    }
    if(i < v2.size()){
      ans[i] += v2[i];
    }
  }
  for(int i = 0; i < ans.size() - 1; i++){
    while(ans[i] >= 10){
      ans[i] -= 10;
      ans[i + 1]++;
    }
  }
  if(ans[ans.size() - 1] < 10){
    return ans;
  }
  ans.push_back(0);
  while(ans[ans.size() - 2] >= 10){
    ans[ans.size() - 2] -= 10;
    ans[ans.size() - 1]++;
  }
  return ans;
}


```

string add(string s1, string s2){
  vector<int> vec1, vec2, ansVec;
  for(int i = s1.size() - 1; i >= 0; i--){
    vec1.push_back(s1[i] - '0');
  }
  for(int i = s2.size() - 1; i >= 0; i--){
    vec2.push_back(s2[i] - '0');
  }
  ansVec = addVec(vec1, vec2);
  reverse(ansVec.begin(), ansVec.end());
  string ansStr;
  for(int i = 0; i < ansVec.size(); i++){
    ansStr += (ansVec[i] + '0');
  }
  return ansStr;
}
